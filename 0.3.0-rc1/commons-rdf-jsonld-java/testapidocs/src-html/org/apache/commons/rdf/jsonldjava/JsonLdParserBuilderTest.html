018package org.apache.commons.rdf.jsonldjava;
019
020import static org.junit.Assert.assertEquals;
021import static org.junit.Assert.assertNotNull;
022import static org.junit.Assert.assertTrue;
023
024import java.io.InputStream;
025import java.net.URL;
026import java.nio.file.Files;
027import java.nio.file.Path;
028import java.nio.file.StandardCopyOption;
029import java.util.concurrent.TimeUnit;
030
031import org.apache.commons.rdf.api.Graph;
032import org.apache.commons.rdf.api.IRI;
033import org.apache.commons.rdf.api.Literal;
034import org.apache.commons.rdf.api.RDFSyntax;
035import org.apache.commons.rdf.jsonldjava.experimental.JsonLdParser;
036import org.apache.commons.rdf.simple.Types;
037import org.junit.Test;
038
039public class JsonLdParserBuilderTest {
040    private static final String TEST_JSONLD = "/test.jsonld";
041
042    static JsonLdRDF factory = new JsonLdRDF();
043    IRI test = factory.createIRI("http://example.com/test");
044    IRI Type = factory.createIRI("http://example.com/Type");
045    IRI type = factory.createIRI("http://www.w3.org/1999/02/22-rdf-syntax-ns#type");
046    IRI pred1 = factory.createIRI("http://example.com/pred1");
047    IRI pred2 = factory.createIRI("http://example.com/pred2");
048    IRI pred3 = factory.createIRI("http://example.com/pred3");
049    IRI pred4 = factory.createIRI("http://example.com/pred4");
050    IRI other = factory.createIRI("http://example.com/other");
051    IRI graph = factory.createIRI("http://example.com/graph");
052
053    @Test
054    public void parseByUrl() throws Exception {
055        URL url = getClass().getResource(TEST_JSONLD);
056        assertNotNull("Test resource not found: " + TEST_JSONLD, url);
057        IRI iri = factory.createIRI(url.toString());
058        Graph g = factory.createGraph();
059        new JsonLdParser().contentType(RDFSyntax.JSONLD).source(iri).target(g).parse().get(10, TimeUnit.SECONDS);
060        checkGraph(g);
061    }
062
063    @Test
064    public void parseByPath() throws Exception {
065        Path path = Files.createTempFile("test", ".jsonld");
066        path.toFile().deleteOnExit();
067        try (InputStream is = getClass().getResourceAsStream(TEST_JSONLD)) {
068            assertNotNull("Test resource not found: " + TEST_JSONLD, is);
069            Files.copy(is, path, StandardCopyOption.REPLACE_EXISTING);
070        }
071        Graph g = factory.createGraph();
072        new JsonLdParser().contentType(RDFSyntax.JSONLD).source(path).target(g).parse().get(10, TimeUnit.SECONDS);
073        checkGraph(g);
074    }
075
076    @Test
077    public void parseByStream() throws Exception {
078        Graph g = factory.createGraph();
079        try (InputStream is = getClass().getResourceAsStream(TEST_JSONLD)) {
080            assertNotNull("Test resource not found: " + TEST_JSONLD, is);
081            new JsonLdParser().base("http://example.com/base/").contentType(RDFSyntax.JSONLD).source(is).target(g)
082                    .parse().get(10, TimeUnit.SECONDS);
083        }
084        checkGraph(g);
085    }
086
087    private void checkGraph(Graph g) {
088        assertTrue(g.contains(test, type, Type));
089        // Should not include statements from the named graph
090
091        assertEquals(1, g.stream(test, pred1, null).count());
092        assertEquals(1, g.stream(test, pred2, null).count());
093
094        assertEquals("Hello", ((Literal) g.stream(test, pred1, null).findFirst().get().getObject()).getLexicalForm());
095        assertTrue(g.contains(test, pred2, other));
096
097        assertEquals("1337", ((Literal) g.stream(test, pred3, null).findFirst().get().getObject()).getLexicalForm());
098        assertEquals(Types.XSD_INTEGER,
099                ((Literal) g.stream(test, pred3, null).findFirst().get().getObject()).getDatatype());
100
101        // While the named graph 'graph' is not included, the relationship
102        // to that @id is included.
103        assertTrue(g.contains(test, pred4, graph));
104    }
105}
