050/**
051 * Apache Jena RDF implementation.
052 * <p>
053 * Instances of JenaRDF can also convert existing objects from Jena with methods
054 * like {@link #asRDFTerm(Node)} and
055 * {@link #asGraph(org.apache.jena.graph.Graph)}, and vice versa from any
056 * Commons RDF object to Jena with the <code>asJena*</code> methods like
057 * {@link #asJenaNode(RDFTerm)} and {@link #asJenaGraph(Graph)}.
058 * <p>
059 * Note that Commons RDF objects created by this class implement the
060 * specializations interfaces like {@link JenaRDFTerm}, {@link JenaGraph} and
061 * {@link JenaTriple}, which provide access to the underlying Jena objects, e.g.
062 * with {@link JenaRDFTerm#asJenaNode()}.
063 * <p>
064 * For the purpose of {@link BlankNode} identity when using
065 * {@link #createBlankNode(String)} (see {@link BlankNode#equals(Object)} and
066 * {@link BlankNode#uniqueReference()}), each instance of JenaRDF uses an
067 * internal random state. If for some reason consistent/reproducible BlankNode
068 * identity is desired, it is possible to retrieve the state as a UUID using
069 * {@link #salt()} for subsequent use with {@link JenaRDF#JenaRDF(UUID)} - note
070 * that such consistency is only guaranteed within the same minor version of
071 * Commons RDF.
072 * 
073 * @see RDF
074 */
075public final class JenaRDF implements RDF {

236    /**
237     * Convert from Jena {@link Node} to any Commons RDF implementation.
238     * <p>
239     * Note that if the {@link Node#isBlank()}, then the factory's
240     * {@link RDF#createBlankNode(String)} will be used, meaning that care
241     * should be taken if reusing an {@link RDF} instance for multiple
242     * conversion sessions.
243     * 
244     * @see #asRDFTerm(Node)
245     * 
246     * @param factory
247     *            {@link RDF} to use for creating {@link RDFTerm}.
248     * @param node
249     *            The Jena Node to adapt. It's {@link Node#isConcrete()} must be
250     *            <code>true</code>.
251     * @return Adapted {@link RDFTerm}
252     * @throws ConversionException
253     *             If the {@link Node} can't be represented as an
254     *             {@link RDFTerm}, e.g. if the node is not concrete or
255     *             represents a variable in Jena.
256     */
257    public static RDFTerm asRDFTerm(RDF factory, Node node) {
258        if (node == null) {
259            return null;
260        }
261        if (factory instanceof JenaRDF) {
262            // No need to convert, just wrap
263            return ((JenaRDF) factory).asRDFTerm(node);
264        }
265        if (node.isURI())
266            return factory.createIRI(node.getURI());
267        if (node.isLiteral()) {
268            String lang = node.getLiteralLanguage();
269            if (lang != null && !lang.isEmpty())
270                return factory.createLiteral(node.getLiteralLexicalForm(), lang);
271            if (node.getLiteralDatatype().equals(XSDDatatype.XSDstring))
272                return factory.createLiteral(node.getLiteralLexicalForm());
273            IRI dt = factory.createIRI(node.getLiteralDatatype().getURI());
274            return factory.createLiteral(node.getLiteralLexicalForm(), dt);
275        }
276        if (node.isBlank())
277            // The factory
278            return factory.createBlankNode(node.getBlankNodeLabel());
279        throw new ConversionException("Node is not a concrete RDF Term: " + node);
280    }

360    /**
361     * Convert from Jena {@link org.apache.jena.graph.Triple} to a Commons RDF
362     * {@link Triple}.
363     * <p>
364     * Note that if any of the triple's nodes {@link Node#isBlank()}, then the
365     * factory's {@link RDF#createBlankNode(String)} will be used, meaning that
366     * care should be taken if reusing an {@link RDF} instance for multiple
367     * conversion sessions.
368     * 
369     * @see #asTriple(org.apache.jena.graph.Triple)
370     *
371     * @param factory
372     *            {@link RDF} to use for creating the {@link Triple} and its
373     *            {@link RDFTerm}s.
374     * @param triple
375     *            Jena triple
376     * @return Converted triple
377     * @throws ConversionException
378     *             if any of the triple's nodes are not concrete or the triple
379     *             is a generalized triple
380     */
381    public static Triple asTriple(RDF factory, org.apache.jena.graph.Triple triple) throws ConversionException {
382        if (factory instanceof JenaRDF) {
383            // No need to convert, just wrap
384            return ((JenaRDF) factory).asTriple(triple);
385        }
386        final BlankNodeOrIRI subject;
387        final IRI predicate;
388        try {
389            subject = (BlankNodeOrIRI) asRDFTerm(factory, triple.getSubject());
390            predicate = (IRI) asRDFTerm(factory, triple.getPredicate());
391        } catch (ClassCastException ex) {
392            throw new ConversionException("Can't convert generalized triple: " + triple, ex);
393        }
394        RDFTerm object = asRDFTerm(factory, triple.getObject());
395        return factory.createTriple(subject, predicate, object);
396    }

505     * Note that if any of the quad's nodes {@link Node#isBlank()}, then the
506     * factory's {@link RDF#createBlankNode(String)} will be used, meaning that
507     * care should be taken if reusing an {@link RDF} instance for multiple
508     * conversion sessions.
509     * 
510     * @see #asQuad(org.apache.jena.sparql.core.Quad)
511     * @see #asGeneralizedQuad(org.apache.jena.sparql.core.Quad)
512     *
513     * @param factory

523    public static Quad asQuad(RDF factory, org.apache.jena.sparql.core.Quad quad) {
524        if (factory instanceof JenaRDF) {
525            // No need to convert, just wrap
526            return ((JenaRDF) factory).asQuad(quad);

528        BlankNodeOrIRI graphName = (BlankNodeOrIRI) (asRDFTerm(factory, quad.getGraph()));
529        BlankNodeOrIRI subject = (BlankNodeOrIRI) (asRDFTerm(factory, quad.getSubject()));
530        IRI predicate = (IRI) (asRDFTerm(factory, quad.getPredicate()));
531        RDFTerm object = asRDFTerm(factory, quad.getObject());
532        return factory.createQuad(graphName, subject, predicate, object);
533    }

631    /**
632     * Convert a CommonsRDF Graph to a Jena Graph. If the Graph was from Jena
633     * originally, return that original object else create a copy using Jena
634     * objects.
635     * 
636     * @param graph
637     *            Commons RDF {@link Graph} to convert
638     * @return Converted Jena {@link org.apache.jena.graph.Graph}
639     */
640    public org.apache.jena.graph.Graph asJenaGraph(Graph graph) {
641        if (graph instanceof JenaGraph)
642            return ((JenaGraph) graph).asJenaGraph();
643        org.apache.jena.graph.Graph g = GraphFactory.createGraphMem();
644        graph.stream().forEach(t -> g.add(asJenaTriple(t)));
645        return g;
646    }

648    /**
649     * Convert a CommonsRDF RDFTerm to a Jena Node. If the RDFTerm was from Jena
650     * originally, return that original object, else create a copy using Jena
651     * objects.
652     * 
653     * @param term
654     *            Commons RDF {@link RDFTerm} to convert
655     * @return Converted Jena {@link Node}
656     */
657    public Node asJenaNode(RDFTerm term) {
658        if (term == null) {
659            return null;
660        }
661        if (term instanceof JenaRDFTerm)
662            // TODO: What if it's a JenaBlankNodeImpl with
663            // a different salt? Do we need to rewrite the
664            // jena blanknode identifier?
665            return ((JenaRDFTerm) term).asJenaNode();
666
667        if (term instanceof IRI)
668            return NodeFactory.createURI(((IRI) term).getIRIString());
669
670        if (term instanceof Literal) {
671            Literal lit = (Literal) term;
672            RDFDatatype dt = NodeFactory.getType(lit.getDatatype().getIRIString());
673            String lang = lit.getLanguageTag().orElse("");
674            return NodeFactory.createLiteral(lit.getLexicalForm(), lang, dt);
675        }
676
677        if (term instanceof BlankNode) {
678            String id = ((BlankNode) term).uniqueReference();
679            return NodeFactory.createBlankNode(id);
680        }
681        throw new ConversionException("Not a concrete RDF Term: " + term);
682    }

684    /**
685     * Convert a CommonsRDF {@link Triple} to a Jena
686     * {@link org.apache.jena.graph.Triple}.
687     * <p>
688     * If the triple was from Jena originally, return that original object, else
689     * create a copy using Jena objects.
690     * 
691     * @param triple
692     *            Commons RDF {@link Triple} to convert
693     * @return Converted Jena {@link org.apache.jena.graph.Triple}
694     */
695    public org.apache.jena.graph.Triple asJenaTriple(Triple triple) {
696        if (triple instanceof JenaTriple)
697            return ((JenaTriple) triple).asJenaTriple();
698        return org.apache.jena.graph.Triple.create(asJenaNode(triple.getSubject()), asJenaNode(triple.getPredicate()),
699                asJenaNode(triple.getObject()));
700    }

703     * Convert a CommonsRDF {@link Quad} to a Jena
704     * {@link org.apache.jena.sparql.core.Quad}.
705     * <p>
706     * If the quad was from Jena originally, return that original object,
707     * otherwise create a copy using Jena objects.
708     *
709     * @param quad
710     *            Commons RDF {@link Quad} to convert
711     * @return Converted Jena {@link org.apache.jena.sparql.core.Quad}
712     */
713    public org.apache.jena.sparql.core.Quad asJenaQuad(Quad quad) {
714        if (quad instanceof JenaQuad) {
715            return ((JenaQuad) quad).asJenaQuad();
716        }
717        return org.apache.jena.sparql.core.Quad.create(asJenaNode(quad.getGraphName().orElse(null)),
718                asJenaNode(quad.getSubject()), asJenaNode(quad.getPredicate()), asJenaNode(quad.getObject()));
719    }

721    // Some simple validations - full IRI parsing is not cheap.
722    private void validateIRI(String iri) {
723        if (iri.contains(" "))
724            throw new IllegalArgumentException();
725        if (iri.contains("<"))
726            throw new IllegalArgumentException();
727        if (iri.contains(">"))
728            throw new IllegalArgumentException();
729    }

732        if (languageTag.contains(" "))
733            throw new IllegalArgumentException("Invalid language tag: " + languageTag);
734    }
