19  package org.apache.commons.rdf.jena.impl;
20  
21  import java.io.StringWriter;
22  import java.util.UUID;
23  import java.util.stream.Stream;
24  
25  import org.apache.commons.rdf.api.BlankNodeOrIRI;
26  import org.apache.commons.rdf.api.IRI;
27  import org.apache.commons.rdf.api.RDFTerm;
28  import org.apache.commons.rdf.api.Triple;
29  import org.apache.commons.rdf.jena.JenaGraph;
30  import org.apache.commons.rdf.jena.JenaRDF;
31  import org.apache.jena.atlas.iterator.Iter;
32  import org.apache.jena.graph.Node;
33  import org.apache.jena.rdf.model.Model;
34  import org.apache.jena.rdf.model.ModelFactory;
35  import org.apache.jena.riot.Lang;
36  import org.apache.jena.riot.RDFDataMgr;
37  
38  class JenaGraphImpl implements JenaGraph {
39  
40      private final org.apache.jena.graph.Graph graph;
41      private final UUID salt;
42      private final transient JenaRDF factory;
43      private Model model;
44  
45      JenaGraphImpl(org.apache.jena.graph.Graph graph, UUID salt) {
46          this.graph = graph;
47          this.salt = salt;
48          this.factory = new JenaRDF(salt);
49      }
50  
51      JenaGraphImpl(Model model, UUID salt) {
52          this.model = model;
53          this.graph = model.getGraph();
54          this.salt = salt;
55          this.factory = new JenaRDF(salt);
56      }
57  
58      @Override
59      public void add(BlankNodeOrIRI subject, IRI predicate, RDFTerm object) {
60          graph.add(org.apache.jena.graph.Triple.create(factory.asJenaNode(subject), factory.asJenaNode(predicate),
61                  factory.asJenaNode(object)));
62      }
63  
64      @Override
65      public void add(Triple triple) {
66          graph.add(factory.asJenaTriple(triple));
67      }
68  
69      @Override
70      public org.apache.jena.graph.Graph asJenaGraph() {
71          return graph;
72      }
73  
74      @Override
75      public void clear() {
76          graph.clear();
77      }
78  
79      @Override
80      public void close() {
81          graph.close();
82      }
83  
84      @Override
85      public boolean contains(BlankNodeOrIRI subject, IRI predicate, RDFTerm object) {
86          return graph.contains(factory.asJenaNode(subject), factory.asJenaNode(predicate), factory.asJenaNode(object));
87      }
88  
89      @Override
90      public boolean contains(Triple triple) {
91          return graph.contains(factory.asJenaTriple(triple));
92      }
93  
94      @Override
95      public void remove(BlankNodeOrIRI subject, IRI predicate, RDFTerm object) {
96          graph.delete(org.apache.jena.graph.Triple.create(factory.asJenaNode(subject), factory.asJenaNode(predicate),
97                  factory.asJenaNode(object)));
98      }
99  
100     @Override
101     public void remove(Triple triple) {
102         graph.delete(factory.asJenaTriple(triple));
103     }
104 
105     @Override
106     public long size() {
107         return graph.size();
108     }
109 
110     @Override
111     public Stream<? extends Triple> stream() {
112         JenaRDF factory = new JenaRDF(salt);
113         return Iter.asStream(graph.find(null, null, null), true).map(factory::asTriple);
114     }
115 
116     @Override
117     public Stream<? extends Triple> stream(BlankNodeOrIRI s, IRI p, RDFTerm o) {
118         JenaRDF factory = new JenaRDF(salt);
119         return Iter.asStream(graph.find(toJenaAny(s), toJenaAny(p), toJenaAny(o)), true).map(factory::asTriple);
120     }
121 
122     private Node toJenaAny(RDFTerm term) {
123         if (term == null)
124             return Node.ANY;
125         return factory.asJenaNode(term);
126     }
127 
128     @Override
129     public String toString() {
130         StringWriter sw = new StringWriter();
131         RDFDataMgr.write(sw, graph, Lang.NT);
132         return sw.toString();
133     }
134 
135     @Override
136     public Model asJenaModel() {
137         if (model == null) {
138             synchronized (this) {
139                 // As Model can be used for locks, we should make sure we don't
140                 // make
141                 // more than one model
142                 if (model == null) {
143                     model = ModelFactory.createModelForGraph(graph);
144                 }
145             }
146         }
147         return model;
148     }
149 
150 }
