018package org.apache.commons.rdf.jsonldjava;
019
020import java.util.List;
021import java.util.Objects;
022import java.util.Optional;
023import java.util.UUID;
024import java.util.function.Predicate;
025import java.util.stream.Stream;
026
027import org.apache.commons.rdf.api.BlankNodeOrIRI;
028import org.apache.commons.rdf.api.GraphLike;
029import org.apache.commons.rdf.api.IRI;
030import org.apache.commons.rdf.api.Literal;
031import org.apache.commons.rdf.api.RDFTerm;
032// NOTE: To avod confusion, don't importing either of the Quad
033import org.apache.commons.rdf.api.Triple;
034import org.apache.commons.rdf.api.TripleLike;
035
036import com.github.jsonldjava.core.RDFDataset;
037import com.github.jsonldjava.core.RDFDataset.Node;
038
039/**
040 * Common abstract {@link GraphLike}.
041 * <p>
042 * Specialised by {@link JsonLdGraph}, {@link JsonLdUnionGraph} and
043 * {@link JsonLdDataset}.
044 *
045 * @param <T>
046 *            specialisation of {@link TripleLike}, e.g. {@link Triple} or
047 *            {@link org.apache.commons.rdf.api.Quad}
048 */
049public interface JsonLdGraphLike<T extends TripleLike> extends GraphLike<T> {
050    /**
051     * Return the underlying JSONLD-Java {@link RDFDataset}.
052     * <p>
053     * Changes in the JSONLD-Java dataset is reflected in this class and vice
054     * versa.
055     * 
056     * @return The underlying JSONLD-JAva RDFDataset
057     */
058    public RDFDataset getRdfDataSet();
059}
060
061abstract class AbstractJsonLdGraphLike<T extends TripleLike> implements JsonLdGraphLike<T> {
062
063    /**
064     * Used by {@link #bnodePrefix()} to get a unique UUID per JVM run
065     */
066    private static UUID SALT = UUID.randomUUID();
067
068    /**
069     * Prefix to use in blank node identifiers
070     */
071    final String bnodePrefix;
072
073    final JsonLdRDF factory;
074
075    /**
076     * The underlying JSON-LD {@link RDFDataset}.
077     * <p>
078     * Note: This is NOT final as it is reset to <code>null</code> by
079     * {@link #close()} (to free memory).
080     */
081    RDFDataset rdfDataSet;
082
083    AbstractJsonLdGraphLike(RDFDataset rdfDataSet) {
084        this(rdfDataSet, "urn:uuid:" + SALT + "#" + "g" + System.identityHashCode(rdfDataSet));
085    }
086
087    AbstractJsonLdGraphLike(RDFDataset rdfDataSet, String bnodePrefix) {
088        this.rdfDataSet = Objects.requireNonNull(rdfDataSet);
089        this.bnodePrefix = Objects.requireNonNull(bnodePrefix);
090        this.factory = new JsonLdRDF(bnodePrefix);
091    }
092
093    AbstractJsonLdGraphLike(String bnodePrefix) {
094        this(new RDFDataset(), bnodePrefix);
095    }
096
097    @Override
098    public void add(T t) {
099        // add triples to default graph by default
100        BlankNodeOrIRI graphName = null;
101        if (t instanceof org.apache.commons.rdf.api.Quad) {
102            org.apache.commons.rdf.api.Quad q = (org.apache.commons.rdf.api.Quad) t;
103            graphName = q.getGraphName().orElse(null);
104        }
105        // FIXME: JSON-LD's rdfDataSet.addQuad method does not support
106        // generalized RDF, so we have to do a naive cast here
107        add(graphName, (BlankNodeOrIRI) t.getSubject(), (IRI) t.getPredicate(), t.getObject());
108    }
109
110    void add(BlankNodeOrIRI graphName, BlankNodeOrIRI subject, IRI predicate, RDFTerm object) {
111        String g = factory.asJsonLdString(graphName);
112        String s = factory.asJsonLdString(subject);
113        String p = factory.asJsonLdString(predicate);
114        if (object instanceof BlankNodeOrIRI) {
115            String o = factory.asJsonLdString((BlankNodeOrIRI) object);
116            rdfDataSet.addQuad(s, p, o, g);
117        } else if (object instanceof Literal) {
118            Literal literal = (Literal) object;
119            String language = literal.getLanguageTag().orElse(null);
120            String datatype = literal.getDatatype().getIRIString();
121            rdfDataSet.addQuad(s, p, literal.getLexicalForm(), datatype, language, g);
122        }
123    }
124
125    public void close() {
126        // Drop the memory reference, but don't clear it
127        rdfDataSet = null;
128    }
129
130    @Override
131    public void clear() {
132        filteredGraphs(null).forEach(s -> s.clear());
133        // In theory we could use
134        // rdfDataSet.clear();
135        // but then we would need to also do
136        // rdfDataSet.put("@default", new ArrayList());
137        // .. both of which seems to be touching too much on JsonLd-Java's
138        // internal structure
139    }
140
141    @Override
142    public boolean contains(T tripleOrQuad) {
143        return stream().anyMatch(Predicate.isEqual(tripleOrQuad));
144    }
145
146    public RDFDataset getRdfDataSet() {
147        return rdfDataSet;
148    }
149
150    @Override
151    public Stream<? extends T> stream() {
152        return rdfDataSet.graphNames().parallelStream().map(rdfDataSet::getQuads)
153                .flatMap(List<RDFDataset.Quad>::parallelStream).map(this::asTripleOrQuad);
154    }
155
156    /**
157     * Convert JsonLd Quad to a Commons RDF {@link Triple} or
158     * {@link org.apache.commons.rdf.api.Quad}
159     * 
160     * 
161     * @see JsonLdRDF#asTriple(Quad)
162     * @see JsonLdRDF#asQuad(Quad)
163     * @param jsonldQuad
164     *            jsonld quad to convert
165     * @return converted {@link TripleLike}
166     */
167    abstract T asTripleOrQuad(RDFDataset.Quad jsonldQuad);
168
169    // This will be made public in JsonLdDataset
170    // and is used by the other methods.
171    boolean contains(Optional<BlankNodeOrIRI> graphName, BlankNodeOrIRI s, IRI p, RDFTerm o) {
172        return filteredGraphs(graphName).flatMap(List::stream).anyMatch(quadFilter(s, p, o));
173    }
174
175    Stream<List<RDFDataset.Quad>> filteredGraphs(Optional<BlankNodeOrIRI> graphName) {
176        return rdfDataSet.graphNames().parallelStream()
177                // if graphName == null (wildcard), select all graphs,
178                // otherwise check its jsonld string
179                // (including @default for default graph)
180                .filter(g -> graphName == null || g.equals(graphName.map(factory::asJsonLdString).orElse("@default")))
181                // remove the quads which match our filter (which could have
182                // nulls as wildcards)
183                .map(rdfDataSet::getQuads);
184    }
185
186    String graphNameAsJsonLdString(T tripleOrQuad) {
187        if (tripleOrQuad instanceof org.apache.commons.rdf.api.Quad) {
188            org.apache.commons.rdf.api.Quad quad = (org.apache.commons.rdf.api.Quad) tripleOrQuad;
189            return quad.getGraphName().map(factory::asJsonLdString).orElse("@default");
190        }
191        return "@default";
192    }
193
194    Predicate<RDFDataset.Quad> quadFilter(BlankNodeOrIRI subject, IRI predicate, RDFTerm object) {
195        Optional<Node> subjectNode = Optional.ofNullable(subject).map(factory::asJsonLdNode);
196        Optional<Node> predicateNode = Optional.ofNullable(predicate).map(factory::asJsonLdNode);
197        Optional<Node> objectNode = Optional.ofNullable(object).map(factory::asJsonLdNode);
198
199        return q -> {
200            if (subjectNode.isPresent() && subjectNode.get().compareTo(q.getSubject()) != 0) {
201                return false;
202            }
203            if (predicateNode.isPresent() && predicateNode.get().compareTo(q.getPredicate()) != 0) {
204                return false;
205            }
206            if (objectNode.isPresent() && objectNode.get().compareTo(q.getObject()) != 0) {
207                return false;
208            }
209            return true;
210        };
211    }
212
213    // NOTE: This is made public in JsonLdDataset and is used by the other
214    // remove methods.
215    void remove(Optional<BlankNodeOrIRI> graphName, BlankNodeOrIRI subject, IRI predicate, RDFTerm object) {
216        // remove the quads which match our filter (which could have nulls as
217        // wildcards)
218        filteredGraphs(graphName).forEach(t -> t.removeIf(quadFilter(subject, predicate, object)));
219    }
