018package org.apache.commons.rdf.api;
019
020import java.util.ConcurrentModificationException;
021import java.util.Iterator;
022import java.util.stream.Stream;
023
024/**
025 * An <a href="http://www.w3.org/TR/rdf11-concepts/#section-rdf-graph"> RDF 1.1
026 * Graph</a>, a set of RDF triples, as defined by
027 * <a href="http://www.w3.org/TR/rdf11-concepts/" >RDF-1.1 Concepts and Abstract
028 * Syntax</a>, a W3C Recommendation published on 25 February 2014.
029 * 
030 * @see RDF#createGraph()
031 */
032public interface Graph extends AutoCloseable, GraphLike<Triple> {
033
034    /**
035     * Add a triple to the graph, possibly mapping any of the components of the
036     * Triple to those supported by this Graph.
037     *
038     * @param triple
039     *            The triple to add
040     */
041    void add(Triple triple);
042
043    /**
044     * Add a triple to the graph, possibly mapping any of the components to
045     * those supported by this Graph.
046     *
047     * @param subject
048     *            The triple subject
049     * @param predicate
050     *            The triple predicate
051     * @param object
052     *            The triple object
053     */
054    void add(BlankNodeOrIRI subject, IRI predicate, RDFTerm object);
055
056    /**
057     * Check if graph contains triple.
058     *
059     * @param triple
060     *            The triple to check.
061     * @return True if the Graph contains the given Triple.
062     */
063    boolean contains(Triple triple);
064
065    /**
066     * Check if graph contains a pattern of triples.
067     *
068     * @param subject
069     *            The triple subject (null is a wildcard)
070     * @param predicate
071     *            The triple predicate (null is a wildcard)
072     * @param object
073     *            The triple object (null is a wildcard)
074     * @return True if the Graph contains any Triples that match the given
075     *         pattern.
076     */
077    boolean contains(BlankNodeOrIRI subject, IRI predicate, RDFTerm object);
078
079    /**
080     * Close the graph, relinquishing any underlying resources.
081     * <p>
082     * For example, this would close any open file and network streams and free
083     * database locks held by the Graph implementation.
084     * <p>
085     * The behaviour of the other Graph methods are undefined after closing the
086     * graph.
087     * <p>
088     * Implementations might not need {@link #close()}, hence the default
089     * implementation does nothing.
090     */
091    @Override
092    default void close() throws Exception {
093    }
094
095    /**
096     * Remove a concrete triple from the graph.
097     *
098     * @param triple
099     *            triple to remove
100     */
101    void remove(Triple triple);
102
103    /**
104     * Remove a concrete pattern of triples from the graph.
105     *
106     * @param subject
107     *            The triple subject (null is a wildcard)
108     * @param predicate
109     *            The triple predicate (null is a wildcard)
110     * @param object
111     *            The triple object (null is a wildcard)
112     */
113    void remove(BlankNodeOrIRI subject, IRI predicate, RDFTerm object);
114
115    /**
116     * Clear the graph, removing all triples.
117     */
118    void clear();
119
120    /**
121     * Number of triples contained by the graph.
122     * <p>
123     * The count of a set does not include duplicates, consistent with the
124     * {@link Triple#equals(Object)} equals method for each {@link Triple}.
125     *
126     * @return The number of triples in the graph
127     */
128    long size();
129
130    /**
131     * Get all triples contained by the graph.<br>
132     * <p>
133     * The iteration does not contain any duplicate triples, as determined by
134     * the {@link Triple#equals(Object)} method for each {@link Triple}.
135     * <p>
136     * The behaviour of the {@link Stream} is not specified if
137     * {@link #add(Triple)}, {@link #remove(Triple)} or {@link #clear()} are
138     * called on the {@link Graph} before it terminates.
139     * <p>
140     * Implementations may throw {@link ConcurrentModificationException} from
141     * Stream methods if they detect a conflict while the Stream is active.
142     * 
143     * @since 0.3.0-incubating
144     * @return A {@link Stream} over all of the triples in the graph
145     */
146    Stream<? extends Triple> stream();
147
148    /**
149     * Get all triples contained by the graph matched with the pattern.
150     * <p>
151     * The iteration does not contain any duplicate triples, as determined by
152     * the {@link Triple#equals(Object)} method for each {@link Triple}.
153     * <p>
154     * The behaviour of the {@link Stream} is not specified if
155     * {@link #add(Triple)}, {@link #remove(Triple)} or {@link #clear()} are
156     * called on the {@link Graph} before it terminates.
157     * <p>
158     * Implementations may throw {@link ConcurrentModificationException} from
159     * Stream methods if they detect a conflict while the Stream is active.
160     * <p>
161     * 
162     * @since 0.3.0-incubating
163     * @param subject
164     *            The triple subject (null is a wildcard)
165     * @param predicate
166     *            The triple predicate (null is a wildcard)
167     * @param object
168     *            The triple object (null is a wildcard)
169     * @return A {@link Stream} over the matched triples.
170     */
171    Stream<? extends Triple> stream(BlankNodeOrIRI subject, IRI predicate, RDFTerm object);
172
173    /**
174     * This method is deprecated, use the equivalent method {@link #stream()}
175     * instead.
176     * 
177     * @return A {@link Stream} over all triples.
178     */
179    @Deprecated
180    default Stream<? extends Triple> getTriples() {
181        return stream();
182    }
183
184    /**
185     * This method is deprecated, use the equivalent method
186     * {@link #stream(BlankNodeOrIRI, IRI, RDFTerm)} instead.
187     * 
188     * @param subject
189     *            The triple subject (null is a wildcard)
190     * @param predicate
191     *            The triple predicate (null is a wildcard)
192     * @param object
193     *            The triple object (null is a wildcard)
194     * @return A {@link Stream} over the matched triples.
195     */
196    @Deprecated
197    default Stream<? extends Triple> getTriples(BlankNodeOrIRI subject, IRI predicate, RDFTerm object) {
198        return stream(subject, predicate, object);
199    }
200
201    /**
202     * Get an Iterable for iterating over all triples in the graph.
203     * <p>
204     * This method is meant to be used with a Java for-each loop, e.g.:
205     * 
206     * <pre>
207     * for (Triple t : graph.iterate()) {
208     *     System.out.println(t);
209     * }
210     * </pre>
211     * 
212     * The behaviour of the iterator is not specified if {@link #add(Triple)},
213     * {@link #remove(Triple)} or {@link #clear()}, are called on the
214     * {@link Graph} before it terminates. It is undefined if the returned
215     * {@link Iterator} supports the {@link Iterator#remove()} method.
216     * <p>
217     * Implementations may throw {@link ConcurrentModificationException} from
218     * Iterator methods if they detect a concurrency conflict while the Iterator
219     * is active.
220     * <p>
221     * The {@link Iterable#iterator()} must only be called once, that is the
222     * Iterable must only be iterated over once. A {@link IllegalStateException}
223     * may be thrown on attempt to reuse the Iterable.
224     *
225     * @return A {@link Iterable} that returns {@link Iterator} over all of the
226     *         triples in the graph
227     * @throws IllegalStateException
228     *             if the {@link Iterable} has been reused
229     * @throws ConcurrentModificationException
230     *             if a concurrency conflict occurs while the Iterator is
231     *             active.
232     */
233    @SuppressWarnings("unchecked")
234    default Iterable<Triple> iterate() throws ConcurrentModificationException, IllegalStateException {
235        return ((Stream<Triple>) stream())::iterator;
236    }
237
238    /**
239     * Get an Iterable for iterating over the triples in the graph that match
240     * the pattern.
241     * <p>
242     * This method is meant to be used with a Java for-each loop, e.g.:
243     * 
244     * <pre>
245     * IRI alice = factory.createIRI("http://example.com/alice");
246     * IRI knows = factory.createIRI("http://xmlns.com/foaf/0.1/");
247     * for (Triple t : graph.iterate(alice, knows, null)) {
248     *     System.out.println(t.getObject());
249     * }
250     * </pre>
251     * <p>
252     * The behaviour of the iterator is not specified if {@link #add(Triple)},
253     * {@link #remove(Triple)} or {@link #clear()}, are called on the
254     * {@link Graph} before it terminates. It is undefined if the returned
255     * {@link Iterator} supports the {@link Iterator#remove()} method.
256     * <p>
257     * Implementations may throw {@link ConcurrentModificationException} from
258     * Iterator methods if they detect a concurrency conflict while the Iterator
259     * is active.
260     * <p>
261     * The {@link Iterable#iterator()} must only be called once, that is the
262     * Iterable must only be iterated over once. A {@link IllegalStateException}
263     * may be thrown on attempt to reuse the Iterable.
264     *
265     * @param subject
266     *            The triple subject (null is a wildcard)
267     * @param predicate
268     *            The triple predicate (null is a wildcard)
269     * @param object
270     *            The triple object (null is a wildcard)
271     * @return A {@link Iterable} that returns {@link Iterator} over the
272     *         matching triples in the graph
273     * @throws IllegalStateException
274     *             if the {@link Iterable} has been reused
275     * @throws ConcurrentModificationException
276     *             if a concurrency conflict occurs while the Iterator is
277     *             active.
278     */
279    @SuppressWarnings("unchecked")
280    default Iterable<Triple> iterate(BlankNodeOrIRI subject, IRI predicate, RDFTerm object)
281            throws ConcurrentModificationException, IllegalStateException {
282        return ((Stream<Triple>) stream(subject, predicate, object))::iterator;
283    }
284}
