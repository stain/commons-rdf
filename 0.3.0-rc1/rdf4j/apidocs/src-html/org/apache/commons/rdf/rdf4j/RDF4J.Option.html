053/**
054 * RDF4J implementation of RDF.
055 * <p>
056 * The {@link #RDF4J()} constructor uses a {@link SimpleValueFactory} to create
057 * corresponding RDF4J {@link Value} instances. Alternatively, this factory can
058 * be constructed with a different {@link ValueFactory} using
059 * {@link #RDF4J(ValueFactory)}.
060 * <p>
061 * {@link #asRDFTerm(Value)} can be used to convert any RDF4J {@link Value} to
062 * an RDFTerm. Note that adapted {@link BNode}s are considered equal if they are
063 * converted with the same {@link RDF4J} instance and have the same
064 * {@link BNode#getID()}.
065 * <p>
066 * {@link #createGraph()} creates a new Graph backed by {@link LinkedHashModel}.
067 * To use other models, see {@link #asGraph(Model)}.
068 * <p>
069 * To adapt a RDF4J {@link Repository} as a {@link Dataset} or {@link Graph},
070 * use {@link #asDataset(Repository, Option...)} or
071 * {@link #asGraph(Repository, Option...)}.
072 * <p>
073 * {@link #asTriple(Statement)} can be used to convert a RDF4J {@link Statement}
074 * to a Commons RDF {@link Triple}, and equivalent {@link #asQuad(Statement)} to
075 * convert a {@link Quad}.
076 * <p>
077 * To convert any {@link Triple} or {@link Quad} to to RDF4J {@link Statement},
078 * use {@link #asStatement(TripleLike)}. This recognises previously converted
079 * {@link RDF4JTriple}s and {@link RDF4JQuad}s without re-converting their
080 * {@link RDF4JTripleLike#asStatement()}.
081 * <p>
082 * Likewise, {@link #asValue(RDFTerm)} can be used to convert any Commons RDF
083 * {@link RDFTerm} to a corresponding RDF4J {@link Value}. This recognises
084 * previously converted {@link RDF4JTerm}s without re-converting their
085 * {@link RDF4JTerm#asValue()}.
086 * <p>
087 * For the purpose of {@link BlankNode} equivalence, this factory contains an
088 * internal {@link UUID} salt that is used by adapter methods like
089 * {@link #asQuad(Statement)}, {@link #asTriple(Statement)},
090 * {@link #asRDFTerm(Value)} as well as {@link #createBlankNode(String)}. As
091 * RDF4J {@link BNode} instances from multiple repositories or models may have
092 * the same {@link BNode#getID()}, converting them with the above methods might
093 * cause accidental {@link BlankNode} equivalence. Note that the {@link Graph}
094 * and {@link Dataset} adapter methods like
095 * {@link #asDataset(Repository, Option...)} and
096 * {@link #asGraph(Repository, Option...)} therefore uses a unique {@link RDF4J}
097 * internally.
098 * 
099 * @see RDF
100 *
101 */
102public final class RDF4J implements RDF {

208     * {@link org.eclipse.rdf4j.model.BNode} is converted to a

229     *             if the value is not a BNode, Literal or IRI

235    /**
236     * Adapt a RDF4J {@link Value} as a Commons RDF {@link RDFTerm}.
237     * <p>
238     * The value will be of the same kind as the term, e.g. a
239     * {@link org.eclipse.rdf4j.model.BNode} is converted to a
240     * {@link org.apache.commons.rdf.api.BlankNode}, a
241     * {@link org.eclipse.rdf4j.model.IRI} is converted to a
242     * {@link org.apache.commons.rdf.api.IRI} and a
243     * {@link org.eclipse.rdf4j.model.Literal}. is converted to a
244     * {@link org.apache.commons.rdf.api.Literal}
245     *
246     * @param value
247     *            The RDF4J {@link Value} to convert.
248     * @param salt
249     *            A {@link UUID} salt to use for uniquely mapping any
250     *            {@link BNode}s. The salt should typically be the same for
251     *            multiple statements in the same {@link Repository} or
252     *            {@link Model} to ensure {@link BlankNode#equals(Object)} and
253     *            {@link BlankNode#uniqueReference()} works as intended.
254     * @return A {@link RDFTerm} that corresponds to the RDF4J value
255     * @throws IllegalArgumentException
256     *             if the value is not a BNode, Literal or IRI
257     */
258    public static RDF4JTerm asRDFTerm(final Value value, UUID salt) {
259        if (value instanceof BNode) {
260            return rdf4j.createBlankNodeImpl((BNode) value, salt);
261        }
262        if (value instanceof org.eclipse.rdf4j.model.Literal) {
263            return rdf4j.createLiteralImpl((org.eclipse.rdf4j.model.Literal) value);
264        }
265        if (value instanceof org.eclipse.rdf4j.model.IRI) {
266            return rdf4j.createIRIImpl((org.eclipse.rdf4j.model.IRI) value);
267        }
268        throw new IllegalArgumentException("Value is not a BNode, Literal or IRI: " + value.getClass());
269    }

353    /**
354     * Adapt an RDF4J {@link Repository} as a Commons RDF {@link Graph}.
355     * <p>
356     * The graph will include triples in the specified contexts.
357     * <p>
358     * Changes to the graph are reflected in the repository, and vice versa.
359     * Triples added/removed to the graph are reflected in all the specified
360     * contexts.
361     * <p>
362     * <strong>Note:</strong> Some operations on the {@link RDF4JGraph} requires
363     * the use of try-with-resources to close underlying
364     * {@link RepositoryConnection}s, including {@link RDF4JGraph#iterate()} and
365     * {@link RDF4JGraph#stream()}.
366     *
367     * @param repository
368     *            RDF4J {@link Repository} to connect to.
369     * @param contexts
370     *            A {@link Set} of {@link BlankNodeOrIRI} specifying the graph
371     *            names to use as a context. The set may include the value
372     *            <code>null</code> to indicate the default graph. The empty set
373     *            indicates any context, e.g. the <em>union graph</em>.
374     * @param option
375     *            Zero or more {@link Option}s
376     * @return A {@link Graph} backed by the RDF4J repository.
377     */
378    public RDF4JGraph asGraph(Repository repository, Set<? extends BlankNodeOrIRI> contexts, Option... option) {
379        EnumSet<Option> opts = optionSet(option);
380        /** NOTE: asValue() deliberately CAN handle <code>null</code> */
381        Resource[] resources = contexts.stream().map(g -> (Resource) asValue(g)).toArray(Resource[]::new);
382        return rdf4j.createRepositoryGraphImpl(Objects.requireNonNull(repository),
383                opts.contains(Option.handleInitAndShutdown), opts.contains(Option.includeInferred), resources);
384    }

386    /**
387     * Adapt a Commons RDF {@link Triple} or {@link Quad} as a RDF4J
388     * {@link Statement}.
389     * <p>
390     * If the <code>tripleLike</code> argument is an {@link RDF4JTriple} or a
391     * {@link RDF4JQuad}, then its {@link RDF4JTripleLike#asStatement()} is
392     * returned as-is. Note that this means that a {@link RDF4JTriple} would
393     * preserve its {@link Statement#getContext()}, and that any
394     * {@link BlankNode}s would be deemed equivalent in RDF4J if they have the
395     * same {@link BNode#getID()}.
396     *
397     * @param tripleLike
398     *            A {@link Triple} or {@link Quad} to adapt
399     * @return A corresponding {@link Statement}
400     */
401    public Statement asStatement(TripleLike tripleLike) {
402        if (tripleLike instanceof RDF4JTripleLike) {
403            // Return original statement - this covers both RDF4JQuad and
404            // RDF4JTriple
405            return ((RDF4JTripleLike) tripleLike).asStatement();
406        }
407
408        org.eclipse.rdf4j.model.Resource subject = (org.eclipse.rdf4j.model.Resource) asValue(tripleLike.getSubject());
409        org.eclipse.rdf4j.model.IRI predicate = (org.eclipse.rdf4j.model.IRI) asValue(tripleLike.getPredicate());
410        Value object = asValue(tripleLike.getObject());
411
412        org.eclipse.rdf4j.model.Resource context = null;
413        if (tripleLike instanceof Quad) {
414            Quad quad = (Quad) tripleLike;
415            context = (org.eclipse.rdf4j.model.Resource) asValue(quad.getGraphName().orElse(null));
416        }
417
418        return getValueFactory().createStatement(subject, predicate, object, context);
419    }

442    /**
443     * Adapt a Commons RDF {@link RDFTerm} as a RDF4J {@link Value}.
444     * <p>
445     * The value will be of the same kind as the term, e.g. a
446     * {@link org.apache.commons.rdf.api.BlankNode} is converted to a
447     * {@link org.eclipse.rdf4j.model.BNode}, a
448     * {@link org.apache.commons.rdf.api.IRI} is converted to a
449     * {@link org.eclipse.rdf4j.model.IRI} and a
450     * {@link org.apache.commons.rdf.api.Literal} is converted to a
451     * {@link org.eclipse.rdf4j.model.Literal}.
452     * <p>
453     * If the provided {@link RDFTerm} is <code>null</code>, then the returned
454     * value is <code>null</code>.
455     * <p>
456     * If the provided term is an instance of {@link RDF4JTerm}, then the
457     * {@link RDF4JTerm#asValue()} is returned without any conversion. Note that
458     * this could mean that a {@link Value} from a different kind of
459     * {@link ValueFactory} could be returned.
460     *
461     * @param term
462     *            RDFTerm to adapt to RDF4J Value
463     * @return Adapted RDF4J {@link Value}
464     */
465    public Value asValue(RDFTerm term) {
466        if (term == null) {
467            return null;
468        }
469        if (term instanceof RDF4JTerm) {
470            // One of our own - avoid converting again.
471            // (This is crucial to avoid double-escaping in BlankNode)
472            return ((RDF4JTerm) term).asValue();
473        }
474        if (term instanceof org.apache.commons.rdf.api.IRI) {
475            org.apache.commons.rdf.api.IRI iri = (org.apache.commons.rdf.api.IRI) term;
476            return getValueFactory().createIRI(iri.getIRIString());
477        }
478        if (term instanceof org.apache.commons.rdf.api.Literal) {
479            org.apache.commons.rdf.api.Literal literal = (org.apache.commons.rdf.api.Literal) term;
480            String label = literal.getLexicalForm();
481            if (literal.getLanguageTag().isPresent()) {
482                String lang = literal.getLanguageTag().get();
483                return getValueFactory().createLiteral(label, lang);
484            }
485            org.eclipse.rdf4j.model.IRI dataType = (org.eclipse.rdf4j.model.IRI) asValue(literal.getDatatype());
486            return getValueFactory().createLiteral(label, dataType);
487        }
488        if (term instanceof BlankNode) {
489            // This is where it gets tricky to support round trips!
490            BlankNode blankNode = (BlankNode) term;
491            // FIXME: The uniqueReference might not be a valid BlankNode
492            // identifier..
493            // does it have to be in RDF4J?
494            return getValueFactory().createBNode(blankNode.uniqueReference());
495        }
496        throw new IllegalArgumentException("RDFTerm was not an IRI, Literal or BlankNode: " + term.getClass());
497    }
