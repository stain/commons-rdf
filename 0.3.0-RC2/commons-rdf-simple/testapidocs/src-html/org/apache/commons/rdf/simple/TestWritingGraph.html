022import java.nio.charset.StandardCharsets;

030import org.apache.commons.rdf.api.Graph;
031import org.apache.commons.rdf.api.IRI;

038/**
039 * Test writing graph
040 */
041public class TestWritingGraph {
042
043    /*
044     * 200k triples should do - about 7 MB on disk. Override with
045     * -Dtriples=20000000 to exercise your memory banks.
046     */
047    private static final long TRIPLES = Long.getLong("triples", 200000L);
048
049    /**
050     * Run tests with -Dkeepfiles=true to inspect /tmp files *
051     */
052    private static boolean KEEP_FILES = Boolean.getBoolean("keepfiles");
053
054    private static Graph graph;
055
056    private static RDF factory;
057
058    @BeforeClass
059    public static void createGraph() throws Exception {
060        factory = new SimpleRDF();
061        graph = factory.createGraph();
062        IRI subject = factory.createIRI("subj");
063        IRI predicate = factory.createIRI("pred");
064        List<IRI> types = new ArrayList<>(Types.values());
065        // Ensure we don't try to create a literal with rdf:langString but
066        // without a language tag
067        types.remove(Types.RDF_LANGSTRING);
068        Collections.shuffle(types);
069        for (int i = 0; i < TRIPLES; i++) {
070            if (i % 11 == 0) {
071                graph.add(subject, predicate, factory.createBlankNode("Example " + i));
072            } else if (i % 5 == 0) {
073                graph.add(subject, predicate, factory.createLiteral("Example " + i, "en"));
074            } else if (i % 3 == 0) {
075                graph.add(subject, predicate, factory.createLiteral("Example " + i, types.get(i % types.size())));
076            } else {
077                graph.add(subject, predicate, factory.createLiteral("Example " + i));
078            }
079        }
080    }
081
082    @AfterClass
083    public static void tearDownClass() throws Exception {
084        graph.clear();
085        graph = null;
086    }
087
088    @Test
089    public void createGraphTiming() throws Exception {
090        createGraph();
091    }
092
093    @Test
094    public void countQuery() {
095        IRI subject = factory.createIRI("subj");
096        IRI predicate = factory.createIRI("pred");
097        long count = graph.stream(subject, predicate, null).unordered().parallel().count();
098        // System.out.println("Counted - " + count);
099        assertEquals(count, TRIPLES);
100    }
101
102    public static String tripleAsString(Triple t) {
103        return t.getSubject().ntriplesString() + " " + t.getPredicate().ntriplesString() + " "
104                + t.getObject().ntriplesString() + " .";
105    }
106
107    @Test
108    public void writeGraphFromStream() throws Exception {
109        Path graphFile = Files.createTempFile("graph", ".nt");
110        if (KEEP_FILES) {
111            System.out.println("From stream: " + graphFile);
112        } else {
113            graphFile.toFile().deleteOnExit();
114        }
115
116        Stream<CharSequence> stream = graph.stream().map(TestWritingGraph::tripleAsString);
117        Files.write(graphFile, stream::iterator, StandardCharsets.UTF_8);
118    }
119
120    @Test
121    public void writeGraphFromStreamFiltered() throws Exception {
122        Path graphFile = Files.createTempFile("graph", ".nt");
123        if (KEEP_FILES) {
124            System.out.println("Filtered stream: " + graphFile);
125        } else {
126            graphFile.toFile().deleteOnExit();
127        }
128
129        IRI subject = factory.createIRI("subj");
130        IRI predicate = factory.createIRI("pred");
131        Stream<CharSequence> stream = graph.stream(subject, predicate, null).map(TestWritingGraph::tripleAsString);
132        Files.write(graphFile, stream::iterator, StandardCharsets.UTF_8);
133
134    }
135
136    @Test
137    public void writeGraphFromStreamFilteredNoMatches() throws Exception {
138        Path graphFile = Files.createTempFile("graph-empty-", ".nt");
139        if (KEEP_FILES) {
140            System.out.println("Filtered stream: " + graphFile);
141        } else {
142            graphFile.toFile().deleteOnExit();
143        }
144
145        IRI subject = factory.createIRI("nonexistent");
146        IRI predicate = factory.createIRI("pred");
147        Stream<CharSequence> stream = graph.stream(subject, predicate, null).map(Object::toString);
148        Files.write(graphFile, stream::iterator, StandardCharsets.UTF_8);
149
150    }
