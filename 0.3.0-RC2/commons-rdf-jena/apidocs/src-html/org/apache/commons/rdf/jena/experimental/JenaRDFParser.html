019package org.apache.commons.rdf.jena.experimental;
020
021import java.io.IOException;
022import java.io.InputStream;
023import java.nio.file.Files;
024import java.util.function.Consumer;
025
026import org.apache.commons.rdf.api.IRI;
027import org.apache.commons.rdf.api.QuadLike;
028import org.apache.commons.rdf.api.RDFTerm;
029import org.apache.commons.rdf.api.RDF;
030import org.apache.commons.rdf.api.TripleLike;
031import org.apache.commons.rdf.experimental.RDFParser;
032import org.apache.commons.rdf.jena.JenaGraph;
033import org.apache.commons.rdf.jena.JenaRDF;
034import org.apache.commons.rdf.simple.experimental.AbstractRDFParser;
035import org.apache.jena.graph.Graph;
036import org.apache.jena.riot.Lang;
037import org.apache.jena.riot.RDFDataMgr;
038import org.apache.jena.riot.system.StreamRDF;
039import org.apache.jena.riot.system.StreamRDFLib;
040
041public class JenaRDFParser extends AbstractRDFParser<JenaRDFParser> implements RDFParser {
042
043    private Consumer<TripleLike> generalizedConsumerTriple;
044    private Consumer<QuadLike<RDFTerm>> generalizedConsumerQuad;
045
046    protected RDF createRDFTermFactory() {
047        return new JenaRDF();
048    }
049
050    public JenaRDFParser targetGeneralizedTriple(Consumer<TripleLike> consumer) {
051        JenaRDFParser c = this.clone();
052        c.resetTarget();
053        c.generalizedConsumerTriple = consumer;
054        return c;
055    }
056
057    public JenaRDFParser targetGeneralizedQuad(Consumer<QuadLike<RDFTerm>> consumer) {
058        JenaRDFParser c = this.clone();
059        c.resetTarget();
060        c.generalizedConsumerQuad = consumer;
061        return c;
062    }
063
064    @Override
065    protected void resetTarget() {
066        super.resetTarget();
067        this.generalizedConsumerTriple = null;
068        this.generalizedConsumerQuad = null;
069    }
070
071    @Override
072    protected void parseSynchronusly() throws IOException {
073        StreamRDF dest;
074        JenaRDF jenaRDF = getJenaFactory();
075        if (getTargetGraph().isPresent() && getTargetGraph().get() instanceof JenaGraph) {
076            Graph jenaGraph = ((JenaGraph) getTargetGraph().get()).asJenaGraph();
077            dest = StreamRDFLib.graph(jenaGraph);
078        } else {
079            if (generalizedConsumerQuad != null) {
080                dest = jenaRDF.streamJenaToGeneralizedQuad(generalizedConsumerQuad);
081            } else if (generalizedConsumerTriple != null) {
082                dest = jenaRDF.streamJenaToGeneralizedTriple(generalizedConsumerTriple);
083            } else {
084                dest = JenaRDF.streamJenaToQuad(getRdfTermFactory().get(), getTarget());
085            }
086        }
087
088        Lang lang = getContentTypeSyntax().flatMap(jenaRDF::asJenaLang).orElse(null);
089        String baseStr = getBase().map(IRI::getIRIString).orElse(null);
090
091        if (getSourceIri().isPresent()) {
092            RDFDataMgr.parse(dest, getSourceIri().get().toString(), baseStr, lang, null);
093        } else if (getSourceFile().isPresent()) {
094            try (InputStream s = Files.newInputStream(getSourceFile().get())) {
095                RDFDataMgr.parse(dest, s, baseStr, lang, null);
096            }
097        } else {
098            RDFDataMgr.parse(dest, getSourceInputStream().get(), baseStr, lang, null);
099        }
100    }
101
102    private JenaRDF getJenaFactory() {
103        return (JenaRDF) getRdfTermFactory().filter(JenaRDF.class::isInstance).orElseGet(this::createRDFTermFactory);
104    }
105
106}
