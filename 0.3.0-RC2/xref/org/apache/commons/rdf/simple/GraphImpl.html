18  package org.apache.commons.rdf.simple;
19  
20  import org.apache.commons.rdf.api.*;
21  import org.apache.commons.rdf.simple.SimpleRDF.SimpleRDFTerm;
22  
23  import java.util.HashSet;
24  import java.util.Objects;
25  import java.util.Set;
26  import java.util.function.Predicate;
27  import java.util.stream.Collectors;
28  import java.util.stream.Stream;
29  
30  /**
31   * A simple, memory-based implementation of Graph.
32   * <p>
33   * {@link Triple}s in the graph are kept in a {@link Set}.
34   * <p>
35   * All Stream operations are performed using parallel and unordered directives.
36   */
37  final class GraphImpl implements Graph {
38  
39      private static final int TO_STRING_MAX = 10;
40      private final Set<Triple> triples = new HashSet<Triple>();
41      private final SimpleRDF factory;
42  
43      GraphImpl(SimpleRDF simpleRDF) {
44          this.factory = simpleRDF;
45      }
46  
47      @Override
48      public void add(BlankNodeOrIRI subject, IRI predicate, RDFTerm object) {
49          BlankNodeOrIRI newSubject = (BlankNodeOrIRI) internallyMap(subject);
50          IRI newPredicate = (IRI) internallyMap(predicate);
51          RDFTerm newObject = internallyMap(object);
52          Triple result = factory.createTriple(newSubject, newPredicate, newObject);
53          triples.add(result);
54      }
55  
56      @Override
57      public void add(Triple triple) {
58          triples.add(internallyMap(triple));
59      }
60  
61      private <T extends RDFTerm> RDFTerm internallyMap(T object) {
62          if (object == null || object instanceof SimpleRDFTerm) {
63              // No need to re-map our own objects.
64              // We support null as internallyMap() is also used by the filters,
65              // and the
66              // factory constructors later do null checks
67              return object;
68          }
69          if (object instanceof BlankNode) {
70              BlankNode blankNode = (BlankNode) object;
71              // This guarantees that adding the same BlankNode multiple times to
72              // this graph will generate a local object that is mapped to an
73              // equivalent object, based on the code in the package private
74              // BlankNodeImpl class
75              return factory.createBlankNode(blankNode.uniqueReference());
76          } else if (object instanceof IRI) {
77              IRI iri = (IRI) object;
78              return factory.createIRI(iri.getIRIString());
79          } else if (object instanceof Literal) {
80              Literal literal = (Literal) object;
81              if (literal.getLanguageTag().isPresent()) {
82                  return factory.createLiteral(literal.getLexicalForm(), literal.getLanguageTag().get());
83              } else {
84                  return factory.createLiteral(literal.getLexicalForm(), (IRI) internallyMap(literal.getDatatype()));
85              }
86          } else {
87              throw new IllegalArgumentException("RDFTerm was neither a BlankNode, IRI nor Literal: " + object);
88          }
89      }
90  
91      private Triple internallyMap(Triple triple) {
92          BlankNodeOrIRI newSubject = (BlankNodeOrIRI) internallyMap(triple.getSubject());
93          IRI newPredicate = (IRI) internallyMap(triple.getPredicate());
94          RDFTerm newObject = internallyMap(triple.getObject());
95          // Check if any of the object references changed during the mapping, to
96          // avoid creating a new Triple object if possible
97          if (newSubject == triple.getSubject() && newPredicate == triple.getPredicate()
98                  && newObject == triple.getObject()) {
99              return triple;
100         } else {
101             return factory.createTriple(newSubject, newPredicate, newObject);
102         }
103     }
104 
105     @Override
106     public void clear() {
107         triples.clear();
108     }
109 
110     @Override
111     public boolean contains(BlankNodeOrIRI subject, IRI predicate, RDFTerm object) {
112         return stream(subject, predicate, object).findFirst().isPresent();
113     }
114 
115     @Override
116     public boolean contains(Triple triple) {
117         return triples.contains(internallyMap(triple));
118     }
119 
120     @Override
121     public Stream<Triple> stream() {
122         return triples.parallelStream().unordered();
123     }
124 
125     @Override
126     public Stream<Triple> stream(final BlankNodeOrIRI subject, final IRI predicate, final RDFTerm object) {
127         final BlankNodeOrIRI newSubject = (BlankNodeOrIRI) internallyMap(subject);
128         final IRI newPredicate = (IRI) internallyMap(predicate);
129         final RDFTerm newObject = internallyMap(object);
130 
131         return getTriples(t -> {
132             // Lacking the requirement for .equals() we have to be silly
133             // and test ntriples string equivalance
134             if (subject != null && !t.getSubject().equals(newSubject)) {
135                 return false;
136             }
137             if (predicate != null && !t.getPredicate().equals(newPredicate)) {
138                 return false;
139             }
140             if (object != null && !t.getObject().equals(newObject)) {
141                 return false;
142             }
143             return true;
144         });
145     }
146 
147     private Stream<Triple> getTriples(final Predicate<Triple> filter) {
148         return stream().filter(filter);
149     }
150 
151     @Override
152     public void remove(BlankNodeOrIRI subject, IRI predicate, RDFTerm object) {
153         Stream<Triple> toRemove = stream(subject, predicate, object);
154         for (Triple t : toRemove.collect(Collectors.toList())) {
155             // Avoid ConcurrentModificationException in ArrayList
156             remove(t);
157         }
158     }
159 
160     @Override
161     public void remove(Triple triple) {
162         triples.remove(internallyMap(triple));
163     }
164 
165     @Override
166     public long size() {
167         return triples.size();
168     }
169 
170     @Override
171     public String toString() {
172         String s = stream().limit(TO_STRING_MAX).map(Object::toString).collect(Collectors.joining("\n"));
173         if (size() > TO_STRING_MAX) {
174             return s + "\n# ... +" + (size() - TO_STRING_MAX) + " more";
175         } else {
176             return s;
177         }
178     }
179 
180 }
