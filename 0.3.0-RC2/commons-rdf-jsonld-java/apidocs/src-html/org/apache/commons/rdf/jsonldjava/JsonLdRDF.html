089    public Node asJsonLdNode(RDFTerm term) {
090        if (term instanceof JsonLdBlankNode) {
091            JsonLdBlankNode jsonLdBlankNode = (JsonLdBlankNode) term;
092            if (jsonLdBlankNode.uniqueReference().startsWith(bnodePrefix)) {
093                // Only return blank nodes 'as is' if they have the same prefix
094                return jsonLdBlankNode.asJsonLdNode();
095            }
096        } else if (term instanceof JsonLdTerm) {
097            // non-Bnodes can always be return as-is
098            return ((JsonLdTerm) term).asJsonLdNode();
099        }
100        if (term instanceof IRI) {
101            return new RDFDataset.IRI(((IRI) term).getIRIString());
102        }
103        if (term instanceof BlankNode) {
104            String ref = ((BlankNode) term).uniqueReference();
105            if (ref.startsWith(bnodePrefix)) {
106                // one of our own (but no longer a JsonLdBlankNode),
107                // we can recover the label after our unique prefix
108                return new RDFDataset.BlankNode(ref.replace(bnodePrefix, ""));
109            }
110            // The "foreign" unique reference might not be a valid bnode string,
111            // we'll convert to a UUID
112            UUID uuid = UUID.nameUUIDFromBytes(ref.getBytes(StandardCharsets.UTF_8));
113            return new RDFDataset.BlankNode("_:" + uuid);
114        }
115        if (term instanceof Literal) {
116            Literal literal = (Literal) term;
117            return new RDFDataset.Literal(literal.getLexicalForm(), literal.getDatatype().getIRIString(),
118                    literal.getLanguageTag().orElse(null));
119        }
120        throw new IllegalArgumentException("RDFTerm not instanceof IRI, BlankNode or Literal: " + term);
121    }

270    String asJsonLdString(BlankNodeOrIRI blankNodeOrIRI) {
271        if (blankNodeOrIRI == null) {
272            return null;
273        }
274        if (blankNodeOrIRI instanceof IRI) {
275            return ((IRI) blankNodeOrIRI).getIRIString();
276        } else if (blankNodeOrIRI instanceof BlankNode) {
277            BlankNode blankNode = (BlankNode) blankNodeOrIRI;
278            String ref = blankNode.uniqueReference();
279            if (ref.startsWith(bnodePrefix)) {
280                // One of ours (but possibly not a JsonLdBlankNode) -
281                // we can use the suffix directly
282                return ref.replace(bnodePrefix, "");
283            } else {
284                // Map to unique bnode identifier, e.g.
285                // _:0dbd92ee-ab1a-45e7-bba2-7ade54f87ec5
286                UUID uuid = UUID.nameUUIDFromBytes(ref.getBytes(StandardCharsets.UTF_8));
287                return "_:" + uuid;
288            }
289        } else {
290            throw new IllegalArgumentException("Expected a BlankNode or IRI, not: " + blankNodeOrIRI);
291        }
292    }
293
294    JsonLdTerm asRDFTerm(final Node node, String blankNodePrefix) {
295        if (node == null) {
296            return null; // e.g. default graph
297        }
298        if (node.isIRI()) {
299            return new JsonLdIRIImpl(node);
300        } else if (node.isBlankNode()) {
301            return new JsonLdBlankNodeImpl(node, blankNodePrefix);
302        } else if (node.isLiteral()) {
303            // TODO: Our own JsonLdLiteral
304            if (node.getLanguage() != null) {
305                return createLiteral(node.getValue(), node.getLanguage());
306            } else {
307                return createLiteral(node.getValue(), createIRI(node.getDatatype()));
308            }
309        } else {
310            throw new IllegalArgumentException("Node is neither IRI, BlankNode nor Literal: " + node);
311        }
312    }

314    RDFDataset.Quad createJsonLdQuad(BlankNodeOrIRI graphName, BlankNodeOrIRI subject, IRI predicate, RDFTerm object) {
315        return new RDFDataset.Quad(asJsonLdNode(subject), asJsonLdNode(predicate), asJsonLdNode(object),
316                asJsonLdString(graphName));
317    }
